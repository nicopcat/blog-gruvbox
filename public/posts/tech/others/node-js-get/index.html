配置sendResponse函数

使用sendResponse()函数，根据请求html读取相应文件 通过end()发给用户

  • sendResponse()函数有三个参数：filename, statusCode, response
  • 导入fs(file system)模块
const fs = require('fs');
  • 使用fs.readFile()函数，读取文件。它有两个参数：filename(路径)，callback
  • 这个callback也有两个参数：error, data。error是错误信息，data是返回的内容
const sendResponse = (filename, statusCode, response) => {
  fs.readFile(`./html/${filename}`, (error, data) => {
    if (error) {
			// 设置报错状态码 500
      response.statusCode = 500;
			// 设置请求头 内容类型：文本
      response.setHeader('Content-Type', 'text/plain')
			// 简单的报错信息      
			response.end('internet error')
    } else {
      response.statusCode = statusCode;
      response.setHeader('Content-Type', 'text/html')
      // 发送读取到的data（文件和信息等）
			response.end(data)
    }
  })
}
